677 template<typename T>
678 inline int cMessage::findTag()
679 {
680  if (!tags)
681  return -1;
682  int n = tags->size();
683  for (int i = 0; i < n; i++) {
684  cObject *tag = (*tags)[i];
685  if (typeid(T) == typeid(*tag)) //TODO this could be regular (non-template) method and go out-of-line if it took typeid(T) as parameter
686  return i;
687  }
688  return -1;
689 }
