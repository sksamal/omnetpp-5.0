109  void checkedAdd(const SimTime& x) {
110  // if operands are the same sign but result has different sign ==> overflow
111  bool sameSign = haveSameSign(t, x.t);
112  t += x.t;
113  if (sameSign && !haveSameSign(t, x.t))
114  overflowAdding(x);
115  }
116 
117  void checkedSub(const SimTime& x) {
118  // if operands are different signs and result has same sign as x ==> overflow
119  bool differentSign = !haveSameSign(t, x.t);
120  t -= x.t;
121  if (differentSign && haveSameSign(t, x.t))
122  overflowSubtracting(x);
123  }
